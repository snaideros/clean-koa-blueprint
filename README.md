# Démarrer un nouveau projet IA TDD avec Koa.js

# Création du projet
`mkdir <mon-nouveau-projet> && cd <mon-nouveau-projet>`

# Télécharge l'archive de ton repo template et l'extrait
`curl -L https://github.com/snaideros/clean-koa-template/tarball/main | tar -xz --strip-components=1`

# Lance le setup interne
`bash ./setup.sh`
